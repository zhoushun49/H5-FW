    切换到下个View 
    @context: 值当前View对象
@url: 切换到下个View
@param: 传递到下个View的参数
@callback: 回调函数中，可接收从下个View传回的参数

View.prototype.toNext = function () {
          context.openView('url', 'param', 'callback');
        };
